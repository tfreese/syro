Kotlin:
tasks.named<ProcessResources>("processResources") {
    val map = mapOf(
        "project_description" to (description ?: ""),
        "project_artifactId" to name,
        "project_version" to version.toString()
    )

    filesMatching("application.yml") {
        // expand(map)
        filter(
            mapOf("tokens" to map),
            org.apache.tools.ant.filters.ReplaceTokens::class.java
        )
    }
}
